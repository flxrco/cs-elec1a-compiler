prog:	DEL_CODE_OP class DEL_CODE_CL
	;

class:	KEY_CLASS DEL_BRCKT_OP IDENTIFIER DEL_BRCKT_CL DEL_CURLY_OP class_stmts DEL_CURLY_CL
	 ;

class_stmts:	class_stmt class_stmts
		   | class_stmt
		   ;

class_stmt:	if_stmt
		  | while_stmt
		  | for_stmt
		  | decl
		  ; 

stmts:	stmt stmts
	 |	stmt
	 ;

stmt:	if_stmt
	|	while_stmt
	|	for_stmt
	|	decl
	|	code_blk
	;

code_blk:	DEL_CURLY_OP stmts DEL_CURLY_CL
		;

decl:	KEY_DECLARE COLON KEY_DATATYPE COLON IDENTIFIER OPE_ASSIGN LITERAL SEMICOLON
	;

if_stmt:	KEY_IF conds stmts
			| KEY_IF conds stmts
			;

else_stmt:	KEY_ELSE stmt
		 ;

while_stmt:	KEY_WHILE conds stmts
		  ;

for_stmt:	KEY_FOR
		;
